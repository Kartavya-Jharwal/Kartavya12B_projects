# Function to identify patterns in a list of numbers
def identify_pattern(numbers):
    if len(numbers) < 2:
        return "Not enough numbers to identify a pattern."

    differences = [numbers[i] - numbers[i-1] for i in range(1, len(numbers))]

    is_arithmetic = all(diff == differences[0] for diff in differences)
    is_geometric = all(numbers[i] / numbers[i-1] == numbers[1] / numbers[0] for i in range(1, len(numbers)))
    is_repeating = all(numbers[i] == numbers[i % len(differences)] for i in range(len(numbers)))

    if is_arithmetic:
        return "Arithmetic Sequence"
    elif is_geometric:
        return "Geometric Sequence"
    elif is_repeating:
        return "Repeating Pattern"
    else:
        return "No specific pattern found."

# Function to identify the values of alphabet characters
def identify_alphabet_values(codes):
    values = []

    for code in codes:
        if 'a' <= code <= 'z':
            value = ord(code) - ord('a') + 1
            values.append(value)
        elif 'A' <= code <= 'Z':
            value = ord(code) - ord('A') + 1
            values.append(value)
        else:
            values.append(None)

    return values

# Input a list of numbers or a string with alphabet characters and codes from the user
user_input = input("Enter a list of numbers or a string with alphabet characters and codes: ")

if any(char.isdigit() for char in user_input):
    # If digits are present, treat it as a list of numbers
    numbers = [int(num) for num in user_input.split()]

    # If no digits are present, treat it as a string with alphabet characters and codes

#To plot a square through function using pyplot
import matplotlib.pyplot as plt
def square(x, y):    
    '''
    Objective: To plot a square
    Input Parameters: x, y - lists of x coordinates and y
    coordinates respectively
    Return Value: None
    '''
    
    if (x[1] - x[0]) <= 0:
        return
    plt.plot(x, y, 'ro--')
    
    return square([x[0]+1, x[1]-1, x[2]-1, x[3]+1, x[4]+1],[y[0]+1, y[1]+1, y[2]-1, y[3]-1, y[4]+1])
    
       
def main():
    '''
    Objective: To plot a square based on user input
    Input Parameter: None
    Return Value: None
    '''
    
    size = int(input('Enter size of the square: '))
    x = [0, size, size, 0, 0]
    y = [0, 0, size, size, 0]
    square(x, y)
    plt.title('Square')
    plt.axis([min(x)-1, max(x)+1, min(y)-1, max(y)+1])
    plt.grid()
    plt.show()

    
if __name__ == '__main__':
    main()

import numpy as np
import matplotlib.pyplot as plt
def estcoefficients(x,y):
    n=np.size(x)
    meanx,meany=np.mean(x),np.mean(y)
    sy=np.sum(y*x-n*meany*meanx)
    sx=np.sum(x*x-n*meanx*meanx)
    a=sx/sy
    b=meany-a*meanx
    return(a,b)
def plotregline(x,y,b):
    plt.scatter(x,y,color="r",marker="o",s=20)
    ypred=b[0]+b[1]*x
    plt.plot(x,ypred,color="g")
    plt.xlabel('SIZE')
    plt.ylabel('COST')
    plt.show()
x=np.array([10,20,30,40,50])#independentvariable
y=np.array([400,800,1100,1700,2100])#dependentvariable
b=estcoefficients(x,y)
plotregline(x,y,b)
